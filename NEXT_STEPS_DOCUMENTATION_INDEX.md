### Save these commands to a script

```bash
# Save in ~/scripts/week2.sh
cat > week2.sh << 'EOF'
#!/bin/bash
# Week 2 quick commands

# Database
alias db-start="docker-compose up -d postgres"
alias db-migrate="cd api && pnpm prisma migrate dev"
alias db-studio="cd api && pnpm prisma studio"

# Testing
alias e2e="cd e2e && npx playwright test"
alias load="k6 run e2e/load-tests/scenario-1-ramp-up.js"

# Docker
alias docker-build="docker-compose -f docker-compose.prod.yml build"
alias docker-up="docker-compose -f docker-compose.prod.yml up"

echo "Week 2 commands loaded!"
EOF
chmod +x week2.sh
source week2.sh
```
