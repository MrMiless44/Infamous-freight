```bash

# Set up environment
cp .env.example .env.next-100
nano .env.next-100  # Add: ENCRYPTION_KEY, REDIS_HOST, etc.

# Install dependencies
pnpm add bull ioredis @elastic/elasticsearch speakeasy qrcode socket.io
```
